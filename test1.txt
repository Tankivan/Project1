#include <iostream> // подключение библиотеки ввода/вывода
#include <math.h> // подключение библиотеки математических функций

using namespace std; // использование пространства имён std

int main() // заголовок функции
{ // начало функции main
    /* предусловие: вводится значение n
    постусловие: программа вычисляет сумму конечного ряда */
    int fact, n; // объявление переменных
    double Q = 0; // объявление переменных
    cin >> n; // ввод значения переменной
    for (int k = 1; k <= n; k++) // цикл с предусловием
    { // начало тела цикла
        if (k == 3) // условие
            continue; // переход на следующую итерацию цикла
        fact = 1; // присвоение значения переменной
        for (int i = 1; i <= k; i++) // цикл для вычисления факториала
            fact *= i; // вычисление факториала
        Q += ((pow(k-3, k-5) * (k+7)) / fact); // присвоение значения переменной
    } // конец тела цикла
    cout << "Q = " << Q << endl; // вывод строки и значения переменной
    return 0; // возвращение значения функции
} // конец функции main
